# Использование BlackRabbitMQ из 1С

## Подключение компоненты

```1c
// Создать объект внешней компоненты
ПодключитьВнешнююКомпоненту("ОбщийМакет.BlackRabbitMQ");
Клиент = Новый("AddIn.BlackRabbitMQ");
```

## Базовый сценарий: публикация

```1c
// Подключиться к RabbitMQ
Клиент.Connect(
    "localhost",     // хост
    5672,            // порт
    "guest",         // пользователь
    "guest",         // пароль
    "/",             // vhost
    0,               // skip
    Ложь,            // SSL
    30               // таймаут (сек)
);

// Опубликовать сообщение
Клиент.BasicPublish(
    "myexchange",    // обменник
    "orders.new",    // routing key
    "{""orderId"": 123}", // тело сообщения
    0,               // skip
    Истина,          // persistent
    ""               // дополнительные заголовки (JSON)
);

Клиент = Неопределено; // disconnect
```

## Чтение сообщений (polling, обратная совместимость)

```1c
// Объявить очередь
ИмяОчереди = Клиент.DeclareQueue(
    "myqueue",       // имя
    Ложь,            // passive
    Истина,          // durable
    Ложь,            // exclusive
    Ложь,            // autoDelete
    0,               // maxPriority
    ""               // доп. аргументы (JSON)
);

// Запустить потребителя
Тэг = Клиент.BasicConsume(
    ИмяОчереди,      // очередь
    "",              // consumerId (авто)
    Ложь,            // noAck
    Ложь,            // exclusive
    1,               // prefetchCount
    ""               // доп. аргументы (JSON)
);

// Цикл чтения
Пока Истина Цикл
    Успех = Клиент.BasicConsumeMessage(
        0,            // skip
        Сообщение,    // выход: тело (строка)
        ID,           // выход: deliveryTag (число)
        5000          // таймаут (мс)
    );

    Если НЕ Успех Тогда
        Прервать;     // таймаут — нет сообщений
    КонецЕсли;

    // Обработать сообщение
    ОбработатьСообщение(Сообщение);

    // Подтвердить
    Клиент.BasicAck(ID);
КонецЦикла;

// Отменить потребителя
Клиент.BasicCancel();
```

## Чтение с Reject (НОВОЕ: requeue)

```1c
// В отличие от upstream PinkRabbitMQ,
// BasicReject поддерживает второй параметр — requeue.

Пока Истина Цикл
    Успех = Клиент.BasicConsumeMessage(0, Сообщение, ID, 5000);
    Если НЕ Успех Тогда
        Прервать;
    КонецЕсли;

    Попытка
        ОбработатьСообщение(Сообщение);
        Клиент.BasicAck(ID);
    Исключение
        // Вернуть сообщение в очередь (НОВЫЙ параметр!)
        Клиент.BasicReject(ID, Истина); // requeue = true
    КонецПопытки;
КонецЦикла;
```

## Работа с заголовками (headers)

```1c
// Установить заголовки через JSON при публикации
Клиент.BasicPublish(
    "myexchange",
    "orders.new",
    "{""orderId"": 123}",
    0,
    Истина,
    "{""x-custom-header"": ""myvalue"", ""x-retry-count"": 0}"
);

// Прочитать заголовки полученного сообщения
ЗаголовкиJSON = Клиент.GetHeaders();
// → {"x-custom-header": "myvalue", "x-retry-count": 0}
```

## Свойства сообщения

```1c
// Установить свойства перед публикацией
Клиент.CorrelationId = "corr-12345";
Клиент.MessageId = "msg-67890";
Клиент.ContentType = "application/json";

// Прочитать свойства полученного сообщения
ИдКорреляции = Клиент.CorrelationId;

```

## Обработка ошибок

```1c
Попытка
    Клиент.Connect("localhost", 5672, "guest", "guest", "/", 0, Ложь, 30);
Исключение
    Ошибка = Клиент.GetLastError();
    Сообщить("Ошибка подключения: " + Ошибка);
    Возврат;
КонецПопытки;
```

## Полный пример: общий модуль 1С

```1c
// ОбщийМодуль.RabbitMQ_Клиент

Перем Клиент Экспорт;

Процедура Подключиться(Хост, Порт, Пользователь, Пароль) Экспорт
    ПодключитьВнешнююКомпоненту("ОбщийМакет.BlackRabbitMQ");
    Клиент = Новый("AddIn.BlackRabbitMQ");
    Клиент.Connect(Хост, Порт, Пользователь, Пароль, "/", 0, Ложь, 30);
КонецПроцедуры

Процедура Отключиться() Экспорт
    Клиент = Неопределено;
КонецПроцедуры

Процедура Опубликовать(Обменник, КлючМаршрутизации, Тело,
                        Постоянное = Истина, Заголовки = "") Экспорт
    Клиент.BasicPublish(Обменник, КлючМаршрутизации, Тело,
                         0, Постоянное, Заголовки);
КонецПроцедуры

Процедура ПрочитатьСообщения(ИмяОчереди, Обработчик, Таймаут = 5000) Экспорт
    Клиент.DeclareQueue(ИмяОчереди, Ложь, Истина, Ложь, Ложь, 0, "");
    Клиент.BasicConsume(ИмяОчереди, "", Ложь, Ложь, 1, "");

    Пока Истина Цикл
        Успех = Клиент.BasicConsumeMessage(0, Сообщение, ID, Таймаут);
        Если НЕ Успех Тогда
            Прервать;
        КонецЕсли;

        Попытка
            Обработчик(Сообщение);
            Клиент.BasicAck(ID);
        Исключение
            // rejected — вернуть в очередь
            Клиент.BasicReject(ID, Истина);
        КонецПопытки;
    КонецЦикла;

    Клиент.BasicCancel();
КонецПроцедуры
```

## Отличия от PinkRabbitMQ

| Метод | PinkRabbitMQ | BlackRabbitMQ |
|---|---|---|
| `BasicReject` | `BasicReject(ID)` — без requeue | `BasicReject(ID, Requeue)` — **добавлен** requeue |
| Название компоненты | `"PinkRabbitMQ"` | `"BlackRabbitMQ"` |
| Тело сообщения | `strlen` — обрезает на `\0` | Бинарно-безопасное (`.size()`) |

Миграция существующего кода:
1. Заменить `"PinkRabbitMQ"` → `"BlackRabbitMQ"`
2. Добавить второй параметр `Истина` в вызовы `BasicReject`, где нужен requeue
